$sites_booked = array( );
foreach (@=grid_booking as $row) {
	if ( $row['SELECT'] == 1 ) {
		array_push($sites_booked, $row);
	}
}

$USER_ID = @@USER_LOGGED;
$start_date = @@START_DATE;
$end_date = @@END_DATE;
if (!empty($sites_booked) and count ($sites_booked) !=0) {
	foreach ($sites_booked as $site_booked) {

		executeQuery("INSERT INTO PMT_FOOD_SITE_BOOKING VALUES "."
					 (' ', {$site_booked['ID']}, '{$start_date}', '{$end_date}', '{$USER_ID}')");
	}
	
}
