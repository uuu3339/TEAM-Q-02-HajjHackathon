$sites_canceled=array();
if(is_array(@=site_cancel) and count(@=site_cancel) >0 ){
	foreach(@=site_cancel as $row){
		if($row['SELECT']==1){
		array_push($sites_canceled,$row);	
		}
	}
}
if(!empty($sites_canceled)){
	foreach($sites_canceled as $site_canceled){
		executeQuery("DELETE FROM PMT_FOOD_SITE_BOOKING WHERE SITE_ID= {$site_canceled['ID']} and START_DATE= '{$site_canceled['START_DATE']}' and END_DATE=' {$site_canceled['END_DATE']}' ");
	}
	
}
